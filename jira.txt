using System.Text.Json;

public static List<FadvAccountConfig> LoadFadvAccountConfigs(
    IReadOnlyCollection<(string ConfigName, string ConfigValue)> appConfigs)
{
    var parts =
        appConfigs
            .Where(c => c.ConfigName.StartsWith(
                "FADV.AccountConfig.Part",
                StringComparison.OrdinalIgnoreCase))
            .OrderBy(c => c.ConfigName)
            .ToList();

    if (parts.Count == 0)
    {
        throw new InvalidOperationException("No FADV.AccountConfig.Part entries found.");
    }

    var allConfigs = new List<FadvAccountConfig>();

    var jsonOptions = new JsonSerializerOptions
    {
        PropertyNameCaseInsensitive = true
    };

    foreach (var part in parts)
    {
        if (string.IsNullOrWhiteSpace(part.ConfigValue))
        {
            throw new InvalidOperationException(
                $"Config '{part.ConfigName}' is empty.");
        }

        FadvAccountConfigRoot? parsed =
            JsonSerializer.Deserialize<FadvAccountConfigRoot>(
                part.ConfigValue,
                jsonOptions);

        if (parsed == null || parsed.Accounts.Count == 0)
        {
            throw new InvalidOperationException(
                $"Config '{part.ConfigName}' contains no accounts.");
        }

        allConfigs.AddRange(parsed.Accounts);
    }

    ValidateAccountConfigs(allConfigs);

    return allConfigs;
}
