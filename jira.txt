import { HttpErrorResponse, HttpInterceptorFn } from '@angular/common/http';
import { inject } from '@angular/core';
import { Router } from '@angular/router';
import { catchError, throwError } from 'rxjs';
import { AuthService } from '../services/auth.service'; // Adjust path if needed

export const authInterceptor: HttpInterceptorFn = (req, next) => {
  const router = inject(Router);
  // Optional: inject AuthService to clear user state
  // const authService = inject(AuthService); 

  return next(req).pipe(
    catchError((error: HttpErrorResponse) => {
      
      // ðŸš¨ TRAP: If API says "401 Unauthorized"
      if (error.status === 401) {
        console.warn('Session expired or unauthorized. Redirecting to login...');
        
        // 1. (Optional) Clear token/state
        localStorage.removeItem('token'); 
        // authService.logout(); 

        // 2. Redirect to Login
        router.navigate(['/login']);
      }

      // Propagate the error so the component stops loading
      return throwError(() => error);
    })
  );
};


import { ApplicationConfig } from '@angular/core';
import { provideRouter } from '@angular/router';
import { provideHttpClient, withInterceptors } from '@angular/common/http'; // ðŸ‘ˆ Import this

import { routes } from './app.routes';
import { authInterceptor } from './core/interceptors/auth.interceptor'; // ðŸ‘ˆ Import your interceptor

export const appConfig: ApplicationConfig = {
  providers: [
    provideRouter(routes),
    
    // âœ… Register the interceptor here
    provideHttpClient(withInterceptors([
      authInterceptor
    ]))
  ]
};
