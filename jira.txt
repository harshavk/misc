public sealed class ProcessingScopeConfig
{
    public string RootPath { get; set; } = string.Empty;
    public ExcludeScopeConfig? Exclude { get; set; }
}

public sealed class ExcludeScopeConfig
{
    public List<ExcludedQuarter>? Quarters { get; set; }
}

public sealed class ExcludedQuarter
{
    public int Year { get; set; }
    public string Quarter { get; set; } = string.Empty;
}

using System.Text.Json;

public static ProcessingScopeConfig DeserializeProcessingScope(string json)
{
    if (string.IsNullOrWhiteSpace(json))
    {
        throw new InvalidOperationException("ProcessingScope config value is empty.");
    }

    var options = new JsonSerializerOptions
    {
        PropertyNameCaseInsensitive = true
    };

    ProcessingScopeConfig? config =
        JsonSerializer.Deserialize<ProcessingScopeConfig>(json, options);

    if (config == null || string.IsNullOrWhiteSpace(config.RootPath))
    {
        throw new InvalidOperationException("Invalid ProcessingScope configuration.");
    }

    return config;
}
