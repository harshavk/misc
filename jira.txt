updatePrompt() {
    const selected = this.missingTools
      .filter(x => x.selected)
      .map(x => x.req.tool);

    if (selected.length === 0) {
      this.allyString = "";
      return;
    }

    const toolList = selected.map(s => `• ${s}`).join('\n');

    // ✅ LOGIC: Switch prompt based on user role
    if (this.data.isManager) {
      // MANAGER VIEW: "Grant access to [Employee Name]..."
      this.allyString = `Hi Ally,
Please grant access for new joiner ${this.data.employeeName} (${this.data.employeeId}).
They are joining Project "${this.projectName}".

Please mirror the access rights of their buddy, ${this.buddyName}.

Required Tools:
${toolList}

Ref: Manager Override`;
    } else {
      // EMPLOYEE VIEW: "Grant access to me..."
      this.allyString = `Hi Ally,
I am working on Project "${this.projectName}".
Please grant me the same access rights as my buddy, ${this.buddyName}.

Required Tools:
${toolList}

Ref: Onboarding Mirroring`;
    }
  }
