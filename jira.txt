using System.Text.Json;

public static Dictionary<string, List<FadvAccountConfig>> LoadFadvAccountConfig(
    IReadOnlyCollection<(string ConfigName, string ConfigValue)> appConfigs)
{
    var accountConfigParts =
        appConfigs
            .Where(c => c.ConfigName.StartsWith(
                "FADV.AccountConfig.Part",
                StringComparison.OrdinalIgnoreCase))
            .OrderBy(c => c.ConfigName)
            .ToList();

    if (accountConfigParts.Count == 0)
    {
        throw new InvalidOperationException("No FADV.AccountConfig.Part entries found.");
    }

    var mergedAccounts =
        new Dictionary<string, List<FadvAccountConfig>>(StringComparer.OrdinalIgnoreCase);

    var jsonOptions = new JsonSerializerOptions
    {
        PropertyNameCaseInsensitive = true
    };

    foreach (var part in accountConfigParts)
    {
        FadvAccountConfigRoot? parsed =
            JsonSerializer.Deserialize<FadvAccountConfigRoot>(
                part.ConfigValue,
                jsonOptions);

        if (parsed == null || parsed.Accounts.Count == 0)
        {
            throw new InvalidOperationException(
                $"Config '{part.ConfigName}' contains no accounts.");
        }

        foreach (var accountEntry in parsed.Accounts)
        {
            if (!mergedAccounts.TryGetValue(accountEntry.Key, out var list))
            {
                list = new List<FadvAccountConfig>();
                mergedAccounts[accountEntry.Key] = list;
            }

            list.AddRange(accountEntry.Value);
        }
    }

    return mergedAccounts;
}
